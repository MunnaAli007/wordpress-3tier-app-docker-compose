#!/bin/sh

# Function to check if a number is 2-digit
is_two_digit() {
    if [[ $1 -ge 10 && $1 -le 99 ]]; then
        return 0  # True
    else
        return 1  # False
    fi
}

# Read the first number
read -p "Enter the first 2-digit number: " num1
if ! is_two_digit $num1; then
    echo "Error: $num1 is not a 2-digit number."
    exit 1
fi

# Read the second number
read -p "Enter the second 2-digit number: " num2
if ! is_two_digit $num2; then
    echo "Error: $num2 is not a 2-digit number."
    exit 1
fi

# Calculate the sum
sum=$((num1 + num2))

# Display the result
echo "Sum of $num1 and $num2 is: $sum"
